CREATE TABLE [dbo].[Categores] (
    [Id_Cat]          INT           IDENTITY (1, 1) NOT NULL,
    [Discription_Cat] NVARCHAR (50) NULL,
    PRIMARY KEY CLUSTERED ([Id_Cat] ASC)
);

CREATE TABLE [dbo].[Customer] (
    [Id_Customer] INT            IDENTITY (1, 1) NOT NULL,
    [First_Name]  NVARCHAR (50)  NULL,
    [Last_Name]   NVARCHAR (50)  NULL,
    [Phone]       NVARCHAR (50)  NULL,
    [Email]       NVARCHAR (50)  NULL,
    [Image]       NVARCHAR (MAX) NULL,
    PRIMARY KEY CLUSTERED ([Id_Customer] ASC)
);

CREATE TABLE [dbo].[Orders_details] (
    [Id_Product] NVARCHAR (50) NOT NULL,
    [Id_Order]   INT           NULL,
    [Qte]        INT           NULL
);

CREATE TABLE [dbo].[Orders] (
    [Id_Order]    INT      IDENTITY (1, 1) NOT NULL,
    [Date_Order]  DATETIME NULL,
    [Id_Customer] INT      NULL,
    PRIMARY KEY CLUSTERED ([Id_Order] ASC)
);

CREATE TABLE [dbo].[Product] (
    [Id_Product]    NVARCHAR (50) NOT NULL,
    [Label_Product] NVARCHAR (50) NULL,
    [Qte_Product]   INT           NULL,
    [Price]         NVARCHAR (50) NULL,
    [Image_Product] IMAGE         NULL,
    [Id_Cat]        INT           NULL,
    PRIMARY KEY CLUSTERED ([Id_Product] ASC)
);
